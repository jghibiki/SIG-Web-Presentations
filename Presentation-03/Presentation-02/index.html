git fetch upstream

Merging Upstream Into Your Master

git checkout master
git merge upstream/master

Now push changes to github:

git push origin master

Web Application Concepts

REST APIs

  • API: Application Programmer Interface
    • Generally defined as a way to access a service
  • REST: REpresentational State Transfer
    • Stateless Data Transfer
    • Cacheable
    • Uniform Interface
    • Makes use of HTTP Verbs

HTTP Verbs

  • GET
    • Used for retrieving data from the server
    • Default HTTP Verb used when loading a page in a web browser
  • POST
    • Used for sending data to the server
    • Default HTTP Verb used when submitting a HTML form
  • PUT
    • Used for updating data on the server
  • DELETE
    • Used for deleting data from the server

Typescript: Interfaces


// Interfaces are structural, anything that has the properties
// is compliant with the interface
interface Person {
      name: string;
        // Optional properties, marked with a "?"
          age?: number;
            // And of course functions
              move(): void;
}

Typescript: Implementing Interfaces


// Object that implements the "Person" interface
// Can be treated as a Person since it has the name and move properties
var p: Person = { name: "Bobby", move: () => {} };
// Objects that have the optional property:
var validPerson: Person = { name: "Bobby", age: 42, move: () => {} };
// Is not a person because age is not a number
var invalidPerson: Person = { name: "Bobby", age: true };

Typescript: Inheritance


// Inheritance
class Point3D extends Point {
  constructor(x: number, y: number, public z: number = 0) {
    // Explicit call to the super class constructor is mandatory
    super(x, y); 
  }

  // Overwrite
  dist() {
    var d = super.dist();
    return Math.sqrt(d * d + this.z * this.z);
  }
}
